  recipe(formula = Sepal.Width ~ ., data = iris) %>% 
  step_novel(all_nominal(), -all_outcomes()) %>% 
  step_dummy(all_nominal(), -all_outcomes()) %>% 
  step_zv(all_predictors()) %>% 
  step_normalize(all_predictors(), -all_nominal())

  nearest_neighbor() %>% 
  set_mode("regression") %>% 
  set_engine("kknn")

  workflow() %>% 
  add_recipe(knn_recipe) %>% 
  add_model(knn_model)
